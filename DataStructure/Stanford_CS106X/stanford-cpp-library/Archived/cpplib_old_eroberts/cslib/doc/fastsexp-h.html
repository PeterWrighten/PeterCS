/*
 * File: fastsexp.h
 * ----------------
 * This interface provides the same facilities as "sexp.h" but implements
 * car and cdr as macros for efficiency.  The disadvantage of this approach
 * is that error-checking is reduced, since there is now no checking to see
 * if the arguments are in fact dotted pairs.
 */

#ifndef _fastsexp_h
#define _fastsexp_h

#include "cslib.h"
#include "sexp.h"
#include "sexpcdt.h"

#define car(x) ((x)->SxValue.DtprRep._Car)
#define cdr(x) ((x)->SxValue.DtprRep._Cdr)

#endif
